Enter Backbone.js

O que é?

Backbone.js é uma biblioteca javascript leve e pequena que ajuda a estruturar seu codigo js.

  • MVC*
  • RESTfull
  • Suporte a roteamento com history e manipulação do DOM
  • Event handling
  • Tamanho: 6.3kb, Packed e gzipped
  • Open Source, MIT

Dependências

  • Underscore.js

Opcional

  • json2.js
  • jQuery ou Zepto

Backbone.Model

No Backbone.js, você representa seus dados usando Models.

var Note = Backbone.Model.extend({
    defaults: {
        firstName: "",
        lastName: "",
        text: "",
    },

    initialize: function() {
        /* codigo de inicialização do modelo*/
        console.log("creating a note...");
    },

    author: function() {
        return this.get("firstName") + " " + this.get("lastName");
    }
});

CRUD

Supondo que uma aplicação com o endpoint '/notes' mapeado, rodando e respondendo os verbos HTTP de maneira adequada, a persistência dos modelos é trivial:

> var Note = Backbone.Model.extend({
    // se nao definirmos uma coleção, precisamos dizer a url do recurso
    urlRoot: '/notes'

    initialize: function () {
        this.on("invalid", function (model, error) {
            console.log(error);
        });
    }
});
> note = new Note();
> note.save('text', "Nossa, mas que legal... Notas..."); // gera um POST em '/notes'
> note.save('firstName', 'Paulo') // PUT
> note.destroy({success: function(model, response) {
    console.log("Nota excluida com sucesso!");
}); // DELETE

Backbone.View

As Views do backbone são camadas muito simples e flexiveis. Com elas é possivel separar a lógica, da apresentação dos modelos, nada de html ou css.

var NoteView = Backbone.View.extend({
  tagName:  'li',
  tpl: _.template("<li><p class='author'><%=firstName%></p>" +
                  "<p class='text'><%=text%></p></li>")
  events: {
    'dblclick text': 'edit',
    'blur .edit':   'close'
  },
  render: function() {
    this.$el.html(this.tpl(this.model.toJSON()));
    this.input = this.$('.edit');
    return this;
  },
  edit: function() { ... },
  close: function() { ... },
});

Agnóstico a sistema de templates: _.template, mustache...

Views 2

E aqui uma view que lista as anotações.

var ListView = Backbone.View.extend({

  el: '#notes',

  initialize: function() {
      // preste atenção aqui...
      this.model.bind('change', _.bind(this.render, this));
  },

  render: function(){
      var notes = this.model.get('notes');
      _.each(notes, function(note){
          var noteView = new NoteView({model: note});
          this.$el.append(noteView.render().el);
      }, this);
  }
});

var listView = new ListView({model: notes})

Backbone.Collection

É uma conveniencia para gerenciar grupos de modelos.

var NotesCollection = Backbone.Collection.extend({
    model: Note,
});
var note1 = new Note({id: 1, firstName: 'Paulo', text: "nota 1"}),
    note2 = new Note({id: 2, firstName:"Charlinho", text: "nota 2"}),
    note3 = new Note({id: 3, firstName: "Paulo", text: "nota 3"});
var notes = new NotesCollection([note1, note2, note3]);
console.log(notes.get(2) === note2);
> true
var myNotes = notes.filter(function(note){
    return note.get("firstName") == "Paulo"}
);
console.log(myNotes === [note1, note3])
> true
notes.sync(); // os dados, agora, vão ser persistidos no servidor
notes.on('add', function(note){ console.log(note.get("text"); });
notes.add(new Note({firstName: "Charlinho", text: "Gosto mais de batata"}))
> "Gosto mais de batata"

Backbone.Router

Para a construção de SPA(single page applications), o Backbone.js te dá o Router. Seja por #fragmentos ou urls "reais", a biblioteca te ajuda. Para browsers que nao suportam a api de history, o backbone se encarrega do fallback.

var NoteRouter = Backbone.Router.extend({
    routes: {
        // ex: GET http://localhost/#notes
        'notes': 'getNotes',
        // ex: GET http://localhost/#note/1
        'note/:id': 'getNote',
        ...
        getNote: function(id) {
            console.log("renderiza as notas");
        },

        getNote: function(id) {
            console.log('Veja que legal, ele me passou o *id* do fragment');
        }
    }
})

Backbone.Events

O modulo de eventos pode ser usado em qualquer objeto e te dá a possibilidade de trabalhar com eventos custom.

Backbone.on('evento', function() {console.log('Algum evento ocorreu');});
Backbone.trigger('evento');
> "Algum evento ocorreu"

var meuObjeto = {
    dance: function() {
        // dançando freneticamente como se nao houvesse amanhã
        console.log("Tuntz! Tuntz! Tuntz!");
    }
};
// extendemos meu objeto
_.extend(meuObjeto, Backbone.Events);

meuObjeto.on("music", function(event) {this.dance()}, meuObjeto)
meuObjeto.trigger("music");

Temos o on, off, listenTo, stopListening ...

Não é coincidencia que os metodos de eventos desses modulos são semelhantes aos que ja vimos no Model, Collection, Routes...

Vantagens

Vejamos algumas vantagens de usar o Backbone.js numa aplicacão:

  • Maior manunebilidade do código
  • Desacoplamento de models e views
  • Simplificação de codigo e separação de responsabilidades facilita a criação dos testes
  • Reduz duplicação(DRY)
  • Bem estável! Milestone 1.0 alcançado
  • Muitos plugins
